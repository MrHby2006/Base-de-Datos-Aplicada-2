--EXIT
DECLARE V_CONTADOR NUMBER := 1;
BEGIN
    LOOP --INSTRUCCIONES A REPETIR
        DBMS_OUTPUT.PUT_LINE('HOLA '|| V_CONTADOR||' VECES');
        V_CONTADOR := V_CONTADOR +1;
        EXIT WHEN V_CONTADOR = 11; --PARA PODER SALIR DEL BUCLE, PUEDE SER CON CONDICION O SIN ESTA,
        --SE PUEDE SALIR CON EL EXIT COMO EL WHILE
    END LOOP;
END;

BEGIN
    FOR i IN 1..5 --PRIMERO LA VARIABLE, LUGEGO POR DONDE INICIO Y POR ULTIMO DONDE TERMINA SEPARADOS POR DOS PUNTOS
    LOOP
        DBMS_OUTPUT.PUT_LINE('HOLA '||i||' VECES');
    END LOOP;
END;

--ACTIVIDADES
/*
    CREE UN BLOQUE ANONIMO QUE PERMITA MOSTRAR:
    - NOMBRE
    - SUELDO
    - ID EMPLEADO
    - EMAIL
    - TELEFONO
    DE TODOS LOS EMPLEADOS USANDO UNA SENTENCIA DE REPETICION
*/
DECLARE
    TYPE RECORD_EMPLEADO IS RECORD(
        NOMBRE VARCHAR(30),
        SUELDO NUMBER,
        ID_EMPLEADO NUMBER,
        EMAIL VARCHAR(40),
        TELEFONO VARCHAR(20)
    );
    
    R_EMPLEADO RECORD_EMPLEADO;
    
    V_PRIMER_EMPLEADO NUMBER;
    V_ULTIMO_EMPLEADO NUMBER;
BEGIN
    SELECT MAX(EMPLOYEE_ID), MIN(EMPLOYEE_ID)
    INTO V_ULTIMO_EMPLEADO, V_PRIMER_EMPLEADO
    FROM EMPLOYEES;
    FOR i IN V_PRIMER_EMPLEADO..V_ULTIMO_EMPLEADO
    LOOP
    SELECT FIRST_NAME, SALARY, EMPLOYEE_ID, EMAIL, PHONE_NUMBER
    INTO R_EMPLEADO
    FROM EMPLOYEES
    WHERE EMPLOYEE_ID = i;
    
    DBMS_OUTPUT.PUT_LINE(' ------------------ ');
    DBMS_OUTPUT.PUT_LINE(R_EMPLEADO.ID_EMPLEADO);
    DBMS_OUTPUT.PUT_LINE(R_EMPLEADO.NOMBRE);
    DBMS_OUTPUT.PUT_LINE(R_EMPLEADO.SUELDO);
    DBMS_OUTPUT.PUT_LINE(R_EMPLEADO.EMAIL);
    DBMS_OUTPUT.PUT_LINE(R_EMPLEADO.TELEFONO);
    END LOOP;
END;

/*
    CREE UN BLOQUE ANONIMO QUE PERMITA MOSTRAR:
    - ID
    - NOMBRE
    DE TODOS LOS DEPARTAMENTOS USANDO UNA SENTENCIA DE REPETICION
*/
DECLARE
    TYPE RECORD_DEPARTAMENTOS IS RECORD(
        ID NUMBER,
        NOMBRE VARCHAR(30)
    );
    
    R_DEPA RECORD_DEPARTAMENTOS;
    V_PRIMER_DEPA NUMBER;
    V_ULTIMO_DEPA NUMBER;
BEGIN
    SELECT MAX(DEPARTMENT_ID), MIN(DEPARTMENT_ID)
    INTO V_ULTIMO_DEPA, V_PRIMER_DEPA
    FROM DEPARTMENTS;
    
    WHILE V_PRIMER_DEPA <= V_ULTIMO_DEPA
    LOOP
        SELECT DEPARTMENT_ID, DEPARTMENT_NAME
        INTO R_DEPA
        FROM DEPARTMENTS
        WHERE DEPARTMENT_ID = V_PRIMER_DEPA;
        
        V_PRIMER_DEPA := V_PRIMER_DEPA +10;
        
        DBMS_OUTPUT.PUT_LINE(' ------------------ ');
        DBMS_OUTPUT.PUT_LINE(R_DEPA.ID);
        DBMS_OUTPUT.PUT_LINE(R_DEPA.NOMBRE);
    END LOOP;
END;
