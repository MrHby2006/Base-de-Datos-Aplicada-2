-- MUESTRE AL SUELDO MAS ALTO DENTRO DE LOS EMPLEADOS DE LA EMPRESA

DECLARE V_SUELDO_MAS_ALTO INTEGER;
        V_NOMBRE_EMPLEADO EMPLOYEES.FIRST_NAME%TYPE; -- FORMA MAS SENCILLA DE SABER EL TIPO DE DATO EL CUAL HAY QUE ASIGNAR
        V_SUELDO_AUMENTADO NUMBER;
        C_PORC_AUMENTO NUMBER := 1.15;

BEGIN
SELECT MAX(SALARY), FIRST_NAME
INTO V_SUELDO_MAS_ALTO, V_NOMBRE_EMPLEADO
FROM EMPLOYEES
WHERE SALARY = (SELECT MAX(SALARY)
                FROM EMPLOYEES)
GROUP BY FIRST_NAME;
V_SUELDO_AUMENTADO := TRUNC(V_SUELDO_MAS_ALTO * C_PORC_AUMENTO); -- PARA DECLARAR VARIABLES ENTRE SI, SE HACE CON :=

DBMS_OUTPUT.PUT_LINE('El empleado '||V_NOMBER_EMPLEADO||' tiene el sueldo mas alto'); -- ESTE ES EL PRINTL DE SQL, SE PUEDE CONCATENA ASI
DBMS_OUTPUT.PUT_LINE('Que es: $'||V_SUELDO_MAS_ALTO||' (suertudo)');
BDMS_OUTPUT.PUT_LINE('El sueldo del empleado '|| V_NOMBRE_EMPLEADO||' es de $'||V_SUELDO_AUMENTADO||' (aun mas suertudo)');
END;
