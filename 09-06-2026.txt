CREATE TABLE PAGO_ATENCION_2026 AS (SELECT * FROM PAGO_ATENCION);
SELECT * FROM PAGO_ATENCION
WHERE EXTRACT(YEAR FROM FECHA_PAGO) == EXTRACT(YEAR FROM SYSDATE)-1;

DECLARE
    V_NUEVO_MONTO NUMBER;
    V_CONTADOR NUMBER := 0;
BEGIN
    FOR R_PAGO IN (SELECT * FROM PAGO_ATENCION
            WHERE EXTRACT(YEAR FROM FECHA_PAGO) = EXTRACT(YEAR FROM SYSDATE)-1)
    LOOP
        V_CONTADOR := V_CONTADOR+1;
        V_NUEVO_MONTO := R_PAGO.MONTO_A_CANCELAR * 0.77;
        UPDATE PAGO_ATENCION_2026
        SET MONTO_A_CANCELAR = V_NUEVO_MONTO
        WHERE ATE_ID = R_PAGO.ATE_ID;
        
        UPDATE PAGO_ATENCION_2026
        SET OBS_PAGO = 'DECUENTO APLICADO '||SYSDATE
        WHERE ATE_ID = R_PAGO.ATE_ID;
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('FIN DEL PROCESO');
    DBMS_OUTPUT.PUT_LINE('Se aplico el descuento a: '||V_CONTADOR||' atenciones del año '||EXTRACT(YEAR FROM SYSDATE));
END;

CREATE TABLE PAGO_ATENCION_FASE2 AS(
    SELECT PA.ATE_ID ATE_ID,
           PA.FECHA_VENC_PAGO FECHA_VENC_PAGO,
           PA.FECHA_PAGO FECHA_PAGO,
           PA.MONTO_ATENCION MONTO_ATENCION,
           PA.MONTO_A_CANCELAR MONTO_A_CANCELAR,
           S.TIPO_SAL_ID
           FROM PAGO_ATENCION PA
           INNER JOIN ATENCION A ON A.ATE_ID = PA.ATE_ID
           INNER JOIN ESPECIALIDAD_MEDICO EM ON EM.MED_RUN = A.MED_RUN
           LEFT JOIN PACIENTE P ON P.PAC_RUN = A.PAC_RUN
           LEFT JOIN SALUD S ON S.SAL_ID = P.SAL_ID
           WHERE EXTRACT(YEAR FROM FECHA_VENC_PAGO) = EXTRACT(YEAR FROM SYSDATE)-1
           AND EM.ESP_ID IN (100, 200, 600, 700)
);
DECLARE
    V_DESCUENTO NUMBER;
    C_PORC_DESCUENTO_FONASA_AB NUMBER := 0.1;
    C_PORC_DESCUENTO_FONASA_CD NUMBER := 0.2;
    V_CONTADOR10 NUMBER := 0;
    V_CONTADOR20 NUMBER := 0;
    V_COMENTARIO VARCHAR(50);
BEGIN
    FOR R_ATENCION IN(SELECT PA.ATE_ID ATE_ID,
           PA.FECHA_VENC_PAGO FECHA_VENC_PAGO,
           PA.FECHA_PAGO FECHA_PAGO,
           PA.MONTO_ATENCION MONTO_ATENCION,
           PA.MONTO_A_CANCELAR MONTO_A_CANCELAR,
           S.TIPO_SAL_ID
           FROM PAGO_ATENCION PA
           INNER JOIN ATENCION A ON A.ATE_ID = PA.ATE_ID
           INNER JOIN ESPECIALIDAD_MEDICO EM ON EM.MED_RUN = A.MED_RUN
           LEFT JOIN PACIENTE P ON P.PAC_RUN = A.PAC_RUN
           LEFT JOIN SALUD S ON S.SAL_ID = P.SAL_ID
           WHERE EXTRACT(YEAR FROM FECHA_VENC_PAGO) = EXTRACT(YEAR FROM SYSDATE)-1
           AND EM.ESP_ID IN (100, 200, 600, 700))
    
    LOOP
        IF R_ATENCION.TIPO_SALUD IN (10, 20) THEN
            V_DESCUENTO := R_ATENCION.MONTO_A_CANCELAR * C_PORC_DESCUENTO_FONASA_AB;
            V_CONTADOR10 := V_CONTADOR10 + 1;
            V_COMENTARIO := 'DESCUENTO APLICADO DE 10% CON FECHA '||SYSDATE;
        ELSIF R_ATENCION.TIPO_SALUD IN (30, 40) THEN
            V_DESCUENTO := R_ATENCION.MONTO_A_CANCELAR * C_POR_DESCUENTO_FONASA_CD;
            V_CONTADOR20 := V_CONTADOR20 + 1;
            V_COMENTARIO := 'DESCUENTO APLICADO DE 20% CON FECHA '||SYSDATE;
        ELSE
            V_DESCUENTO := 0;
        END IF;
        
        R_ATENCION.MONTO_A_CANCELAR := R_ATENCION.MONTO_A_CANCELAR - V_DESCUENTO;
        INSERT INTO PAGO_ATENCION_FASE2 VALUES(R_ATENCION.ATE_ID, 
                                               R_ATENCION.FECHA_PAGO, 
                                               R_ATENCION.FECHA_VENC_PAGO,
                                               R_ATENCION.MONTO_ATENCION,
                                               R_ATENCION.MONTO_A_CANCELAR,
                                               V_COMENTARIO);
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('FIN PROCESO');
    DBMS_OUTPUT.PUT_LINE('SE APLICO DESCUENTO DEL 10% A: '||V_CONTADOR10);
    DBMS_OUTPUT.PUT_LINE('SE APLICO DESCUENTO DEL 120% A: '||V_CONTADOR20);
    DBMS_OUTPUT.PUT_LINE('REVISE LA TABLA DE SIMULACION PAGO_ATENCION_FASE2');
END;
