DECLARE

    CURSOR C_EMPLEADO IS --CURSOR ES EL RESULTADO DE UNA QUERY GUARDADA EN UNA VARIABLE
        SELECT E.FIRST_NAME AS NOMBRE, E.LAST_NAME AS APELLIDO, E.SALARY AS SALARIO, D.DEPARTMENT_NAME AS NOMBRE_DEPA
        FROM EMPLOYEES E
        JOIN DEPARTMENTS D ON D.DEPARTMENT_ID = E.DEPARTMENT_ID;
    
    TYPE RECORD_EMPLEADO IS RECORD(
    NOMBRE VARCHAR(20),
    APELLIDO VARCHAR(25),
    SUELDO NUMBER,
    NOMBRE_DEPA VARCHAR(30)
    );
    
    R_EMPLEADO RECORD_EMPLEADO;
    
BEGIN
    
    --EJECUTA LA QUERY DEL CURSOR
    OPEN C_EMPLEADO;
    
    LOOP
        --RECUPERA LA FILA APUNTADA POR EL PUNTERO, Y HACE AVANZAR EL PUNTERO A LA SIGUIENTE FILA
        FETCH C_EMPLEADO INTO R_EMPLEADO;
        
        EXIT WHEN C_EMPLEADO%NOTFOUND; --SI NO ENCUENTRA DATOS CIERRA EL LOOP
        
        DBMS_OUTPUT.PUT_LINE('----------------------------');
        DBMS_OUTPUT.PUT_LINE('NOMBRE: '||R_EMPLEADO.NOMBRE||' '||R_EMPLEADO.APELLIDO);
        DBMS_OUTPUT.PUT_LINE('SUELDO: $'||R_EMPLEADO.SUELDO);
        DBMS_OUTPUT.PUT_LINE('DEPARTAMENTO: '||R_EMPLEADO.NOMBRE_DEPA);
    END LOOP;
    
    --CIERRA EL CURSOR Y LIBERA LA MEMORIA
    CLOSE C_EMPLEADO;
    
END;
