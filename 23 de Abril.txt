ALTER SESSION SET "_ORACLE_SCRIPT" = TRUE

//CREA USUARIOS//
CREATE USER DESA_INF
IDENTIFIED BY 123456
DEFAULT TABLESPACE USERS
TEMPORARY TABLESPACE temp
QUOTA UNLIMITED ON users;

GRANT CREATE SESSION TO DESA_INF; //DA PERMISOS//
GRANT CREATE TABLE TO DESA_INF; //DA PERMISOS PARA QUE USUARIO CREE TABLAS//
GRANT CREATE SEQUENCE TO DESA_INF; //DA PERMISOS PARA QUE USUARIO CREE SEQUENCIAS//
GRANT CREATE VIEW TO DESA_INF; //DA PERMISOS PARA QUE USUARIO CREE VISTAS//

GRANT SELECT ON HR.EMPLOYEES TO DESA_INF; //DA PERMISO PARA HACER SELECT AL ESQUEMA HR Y A LA TABLA EMPLEADOS//
GRANT UPDATE (DEPARTMENT_NAME, LOCATION_ID) ON HR.DEPARTMENTS TO DESA_INF; //DA PERMISOS PARA PODER ACTUALIZAR CIERTAS FILAS DEL ESQUEMA HR, DE LA TABLA DEPARTAMENTOS//
GRANT ALL ON HR.DEPARTMENTS TO DESA_INF; //DA PERMISOS PARA HACER TODO EN EL ESQUEMA HR, DE LA TABLA DEPARTAMENTOS//

---------------------------------------------------
//CREA PERFILES//
CREATE PROFILE NOMBRE_PERFIL LIMIT
SESSIONS_PER_USER numero
CONNECT_TIME minutos
IDLE_TIME minutos
FAILED_LOGIN_ATTEMPTS numero
PASSWORD_LIFE_TIME dias
PASSWORD_LOCK_TIME dias
PASSWORD_GRACE_TIME dias;

//EJEMPLO//
CREATE PROFILE devs LIMIT
SESSIONS_PER_USER UNLIMITED
IDLE_TIME 30
FAILED_LOGIN_ATTEMPTS 3
PASSWORD_LIFE_TIME 15;

---------------------------------------------------
//ALTERAR USUARIO//
ALTER USER DESA_INF PROFILE devs;
ALTER USER DESA_INF IDENTIFIED BY 12345;

---------------------------------------------------
//ALTERAR PERFILES//
ALTER PROFILE devs LIMIT CONNECT_TIME 60;

---------------------------------------------------
//ELIMINAR USUARIOS//
DROP USER DESA_INF2;

---------------------------------------------------
//CREAR ROLES//
CREATE ROLE ROL_GERENCIA;

//AGRUPAR PERMISOS EN UNO SOLO//
GRANT CREATE TABLE, CREATE VIEW TO ROL_GERENCIA;
GRANT SELECT, UPDATE, INSERT ON HR.EMPLOYEES TO ROL_GERENCIA;

//ASIGNAR UN ROL A UN USUARIO AUTOMATICAMENTE//
GRANT ROL_GERENCIA TO DESA_INF;
ALTER USER DESA_INF DEFAULT ROLE "ROL_GERENCIA";

---------------------------------------------------
//REMUEVE PERMISOS A USUARIOS//
REVOKE CREATE TABLE FROM DESA_INF;

//REMUEVE PERMISOS A ROLES//
REVOKE CREATE TABLE FROM ROL_GERENCIA;
