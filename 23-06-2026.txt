SELECT E.ID_TIPO_EMB AS ID_TIPO_EMB,
       TE.NOMBRE_TIPO_EMB AS TIPO_EMB,
       COUNT(DISTINCT E.MATRICULA) AS CANTIDAD_EMBARCACIONES,
       COUNT(AE.ID_ARRIENDO) AS TOTAL_ARRIENDO,
       SUM(AE.DIAS_SOLICITADOS) AS TOTAL_DIAS_ARRIENDO,
       E.VALOR_ARRIENDO_DIA AS VALOR_ARRIENDO_DIARIO
FROM EMBARCACION E
INNER JOIN TIPO_EMBARCACION TE ON E.ID_TIPO_EMB = TE.ID_TIPO_EMB
INNER JOIN ARRIENDO_EMBARCACION AE ON AE.MATRICULA = E.MATRICULA
GROUP BY E.ID_TIPO_EMB, TE.NOMBRE_TIPO_EMB, E.VALOR_ARRIENDO_DIA;

DECLARE
       V_CANT_EMB NUMBER;
       V_TOTAL_ARRIENDOS NUMBER;
       V_TOTAL_DIAS NUMBER;
       V_INGRESO_TOTAL   NUMBER;
       V_CLASIFICACION VARCHAR(6);
BEGIN
       FOR R_RESUMEN IN(SELECT E.ID_TIPO_EMB AS ID_TIPO_EMB,
                               TE.NOMBRE_TIPO_EMB AS TIPO_EMB,
                               COUNT(DISTINCT E.MATRICULA) AS CANTIDAD_EMBARCACIONES,
                               COUNT(AE.ID_ARRIENDO) AS TOTAL_ARRIENDO,
                               SUM(AE.DIAS_SOLICITADOS) AS TOTAL_DIAS_ARRIENDO,
                               E.VALOR_ARRIENDO_DIA AS VALOR_ARRIENDO_DIARIO
                        FROM EMBARCACION E
                        INNER JOIN TIPO_EMBARCACION TE ON E.ID_TIPO_EMB = TE.ID_TIPO_EMB
                        INNER JOIN ARRIENDO_EMBARCACION AE ON AE.MATRICULA = E.MATRICULA
                        GROUP BY E.ID_TIPO_EMB, TE.NOMBRE_TIPO_EMB, E.VALOR_ARRIENDO_DIA)
       LOOP
              SELECT COUNT(*) INTO V_CANT_EMB 
              FROM EMBARCACION 
              WHERE ID_TIPO_EMB = R_RESUMEN.ID_TIPO_EMB;

              SELECT COUNT(A.ID_ARRIENDO), 
                     SUM(A.DIAS_SOLICITADOS), 
                     SUM(A.DIAS_SOLICITADOS * E.VALOR_ARRIENDO_DIA)
              INTO V_TOTAL_ARRIENDOS, V_TOTAL_DIAS, V_INGRESO_TOTAL
              FROM ARRIENDO_EMBARCACION A
              JOIN EMBARCACION E ON A.MATRICULA = E.MATRICULA
              WHERE E.ID_TIPO_EMB = R_RESUMEN.ID_TIPO_EMB
              AND EXTRACT(YEAR FROM A.FECHA_ARRIENDO) = 2025;

              IF V_INGRESO_TOTAL >= 9000000 THEN
                     V_CLASIFICACION := 'Alta';
              ELSIF V_INGRESO_TOTAL >= 5000000 THEN
                     V_CLASIFICACION := 'Media';
              ELSE
                     V_CLASIFICACION := 'Baja';
              END IF;
              INSERT INTO RESUMEN_TIPO_EMBARCACION (ID_TIPO_EMB, ANNO_PROCESO, NOMBRE_TIPO_EMB, 
                                            CANTIDAD_EMBARCACIONES, TOTAL_ARRIENDOS, 
                                            TOTAL_DIAS, INGRESO_TOTAL, CLASIFICACION)
        VALUES (R_RESUMEN.ID_TIPO_EMB, 2025, R_RESUMEN.NOMBRE_TIPO_EMB, 
                V_CANT_EMB, V_TOTAL_ARRIENDOS, V_TOTAL_DIAS, V_INGRESO_TOTAL, V_CLASIFICACION);
       END LOOP;
END;
