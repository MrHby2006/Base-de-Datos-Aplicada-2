/*
    AUMENTO DE SUELDO POR EMPLEADO
    RECORRE TODOS LOS EMPLEADOS DEL DEPARTAMENTO 50, SI UN SUELDO ES MENOR A 3000
    AUMENTALO UN 15%, SI ETA ENTRE 3000 Y 5000 AUMENTALO UN 8%, SI SUPERA LOS
    5000 NO LO TOQUES.
    AL FINAL MUESTRA LA CANTIDAD DE EMPLEADOS POR CADA TRAMO DE SULDO
*/
--------------------------------------------------------------
CREATE TABLE EMPLEADO_RESPALDO AS (SELECT * FROM EMPLOYEES);

DECLARE
    V_CONTADOR_BAJOS   NUMBER := 0;
    V_CONTADOR_MEDIOS  NUMBER := 0;
    V_CONTADOR_ALTOS   NUMBER := 0;
BEGIN
    FOR R_EMPLEADO IN (SELECT EMPLOYEE_ID, SALARY -- NO ES NECESARIO DECLARAS SI ES DENTRO DEL FOR
                        FROM EMPLEADO_RESPALDO    -- HACE SE CREA DE MANERA IMPLICITA
                        WHERE DEPARTMENT_ID = 50)
    LOOP
        IF R_EMPLEADO.SALARY < 3000 THEN
            UPDATE EMPLEADO_RESPALDO
            SET SALARY = SALARY * 1.15
            WHERE EMPLOYEE_ID = R_EMPLEADO.EMPLOYEE_ID;
            
            V_CONTADOR_BAJOS := V_CONTADOR_BAJOS +1;
        ELSIF R_EMPLEADO.SALARY >= 3000 AND R_EMPLEADO.SALARY <= 5000 THEN
            UPDATE EMPLEADO_RESPALDO
            SET SALARY = SALARY * 1.08
            WHERE EMPLOYEE_ID = R_EMPLEADO.EMPLOYEE_ID;
            
            V_CONTADOR_MEDIOS := V_CONTADOR_MEDIOS +1;
        ELSE
            V_CONTADOR_ALTOS := V_CONTADOR_ALTOS +1;
        END IF;
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('EMPLEADO DEL TRAMOS MENOR A 3000: '||V_CONTADOR_BAJOS);
    DBMS_OUTPUT.PUT_LINE('EMPLEADO DEL TRAMOS ENTRE 3000 Y 5000: '||V_CONTADOR_MEDIOS);
    DBMS_OUTPUT.PUT_LINE('EMPLEADO DEL TRAMOS MAYOR A Y000: '||V_CONTADOR_ALTOS);
END;
--------------------------------------------------------------
/*
    CREE UN BLOQUE ANONIMO QUE PERMITA:
    - CREAR UNA TABLA LLAMDA IN_MANAGER (ID, NOMBRE Y APELLIDO, DEPARTAMENTO)
    - DEBE RECORRER TODOS LOS EMPLEADOS BUSCANDO CUYO MANAGER ID ES NULL
    - INSERTA CADA EMPLEADO ENCONTRADO SIN MANAGER A LA TABLA RECIEN CREADA
*/
--------------------------------------------------------------
BEGIN
    BEGIN
    EXECUTE IMMEDIATE 'TRUNCATE TABLE SIN_MANAGER'; --PARA EJECUTAR INMEDIATAMENTE COMO SQL DINAMICO
    EXCEPTION
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('NO SE TRUNCO PORQUE NO TENIA DATOS');
    END;
    FOR R_EMPLEADO IN (SELECT EMPLOYEE_ID, 
                        FIRST_NAME || LAST_NAME NOMBRE,
                        DEPARTMENT_ID 
                        FROM EMPLOYEES  
                        WHERE MANAGER_ID = NULL)
    LOOP
        INSERT INTO SIN_MANAGER VALUES(R_EMPLEADO.EMPLOYEE_ID,
                                        R_EMPLEADO.NOMBRE,
                                        R_EMPLEADO.DEPRTAMENTO_ID);
    END LOOP;
END;
