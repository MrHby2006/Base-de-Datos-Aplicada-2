DECLARE
    V_INGRESOS_ARRIENDO     NUMBER;
    V_INGRESOS_TOTALES      NUMBER;
    V_CLASIFICACION_DEMANDA VARCHAR(25);
BEGIN
    FOR R_EMBARCACION IN (SELECT E.MATRICULA AS MATRICULA,
                                 TE.NOMBRE_TIPO_EMB AS TIPO_EMB,
                                 COUNT(AE.MATRICULA) AS CANTIDAD_ARRIENDOS,
                                 E.VALOR_ARRIENDO_DIA AS VALOR_ARRIENDO_DIA,
                                 SUM(E.VALOR_GARANTIA_DIA) AS SUMA_GARANTIAS_EMB,
                                 SUM(AE.DIAS_SOLICITADOS) AS DIAS_SOLICITADOS
                          FROM EMBARCACION E
                          LEFT JOIN TIPO_EMBARCACION TE ON E.ID_TIPO_EMB = TE.ID_TIPO_EMB
                          LEFT JOIN ARRIENDO_EMBARCACION AE ON AE.MATRICULA = E.MATRICULA
                          WHERE EXTRACT(YEAR FROM AE.FECHA_INI_ARRIENDO) = EXTRACT(YEAR FROM SYSDATE)-1
                          GROUP BY E.MATRICULA, TE.NOMBRE_TIPO_EMB, E.VALOR_ARRIENDO_DIA)
    LOOP
        V_INGRESOS_ARRIENDO := R_EMBARCACION.VALOR_ARRIENDO_DIA * R_EMBARCACION.CANTIDAD_ARRIENDOS;
        V_INGRESOS_TOTALES := R_EMBARCACION.SUMA_GARANTIAS_EMB + V_INGRESOS_ARRIENDO;
        
        IF R_EMBARCACION.CANTIDAD_ARRIENDOS >= 6 THEN
            V_CLASIFICACION_DEMANDA := 'ALTA';
        ELSIF R_EMBARCACION.CANTIDAD_ARRIENDOS >= 3 OR R_EMBARCACION.CANTIDAD_ARRIENDOS <= 5 THEN
            V_CLASIFICACION_DEMANDA := 'MEDIA';
        ELSIF R_EMBARCACION.CANTIDAD_ARRIENDOS >= 1 OR R_EMBARCACION.CANTIDAD_ARRIENDOS <= 2 THEN
            V_CLASIFICACION_DEMANDA := 'BAJA';
        ELSE
            V_CLASIFICACION_DEMANDA := 'SIN DEMANDA';
        END IF;
        
        INSERT INTO INGRESO_EMBARCACION VALUES(R_EMBARCACION.MATRICULA,
                                               EXTRACT(YEAR FROM SYSDATE) - 1,
                                               R_EMBARCACION.TIPO_EMB,
                                               R_EMBARCACION.CANTIDAD_ARRIENDOS,
                                               R_EMBARCACION.DIAS_SOLICITADOS,
                                               V_INGRESOS_ARRIENDO,
                                               R_EMBARCACION.SUMA_GARANTIAS_EMB,
                                               V_INGRESOS_TOTALES,
                                               V_CLASIFICACION_DEMANDA
        );
    END LOOP;
END;

SELECT * FROM INGRESO_EMBARCACION;
