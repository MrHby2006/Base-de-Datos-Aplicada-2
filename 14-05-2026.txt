/*
    ESCRIBA UN BLOQUE PLSQL QUE CONSULTE A LA TABA EMPLEADOS
    POR EL ID DE EMPLEADO 10, PARA CALCULAR LA ANTIGÜEDAD QUE
    TIENE EN LA EMPRESA,
    SI EL EMPLEADO TIENE MAS DE 10 AÑOS EN LA EMPRESA RECIBE UN BONO QUE
    SE DEBE MOSTRAR POR CONSOLA:
    AÑOS DE ANTGÜEDAD : X AÑOS
    SI CUMPLE CON LA ANTIÜEDAD:
    BONO POR ANTIGÜEDAD : 1.000 USD
    SI NO CUMPLE:
    NO APLICA BONO DE ANTIGÜEDAD
*/
DECLARE
    V_FECHA_CONTRATACION EMPLOYEES.HIRE_DATE%TYPE;
    C_ID_EMPLEADO        INTEGER := &ID; -- EL & SIRVE PARA QUE TE PIDAN A TI EL DATO EL CUAL BUSCAS
    V_ANIOS              INTEGER;
BEGIN
    SELECT HIRE_DATE
    INTO V_FECHA_CONTRATACION
    FROM EMPLOYEES
    WHERE EMPLOYEE_ID = C_ID_EMPLEADO;
    
    V_ANIOS := MONTHS_BETWEEN(SYSDATE, V_FECHA_CONTRATACION)/12;
    
    DBMS_OUTPUT.PUT_LINE('AÑOS DE ANTIGÜEDAD: '||V_ANIOS||' AÑOS');
    IF V_ANIOS > 10 THEN
        DBMS_OUTPUT.PUT_LINE('EL BONO POR ANTIGÜEDAD: 1.000 USD');
    ELSE
        DBMS_OUTPUT.PUT_LINE('NO APLICA BONO DE ANTIGÜEDAD');
    END IF;
END;

/*
    CREA UN BLOQUE ANONIMO QUE PERMITA CONSULTAR POR EL SUELDO DE UN EMPLEADO
    SI EL SUELDO ESTA SOBRE 10.000 DEBE MOSTRAR:
    NOMBRE DEL EMPLEATO -> SUELDO ALTO
    SI ESTA EL SUELDO DEL EMPLEADO ENTRE 7.000 Y 9.999
    NOMBRE DEL EMPLEADO -> BUEN SUELDO
    SI ESTA EL SUELDO DEL EMPLEADO ENTRE 6.999 Y 4.000
    NOMBRE DEL EMPLEADO -> SUELDO BAJO
    SI ESTA EL SUELDO DEL EMPLEADO ENTRE 3.999 O INFERIOR
    NOMBRE DEL EMPLEADO -> F
    EL SUELDO DEL EMPLEADODEBE SER COMPUESTO POR NOMBRE Y APELLIDO
*/
DECLARE
   -- V_EMPLEADO_ID   EMPLOYEES.EMPLOEE_ID%TYPE;
    V_NOMBRE        EMPLOYEES.FIRST_NAME%TYPE;
    V_APELLIDO      EMPLOYEES.LAST_NAME%TYPE;
    V_SUELDO        EMPLOYEES.SALARY%TYPE;
    V_MENSAJE       VARCHAR2(100);
BEGIN
    SELECT FIRST_NAME, LAST_NAME, SALARY
    INTO V_NOMBRE, V_APELLIDO, V_SUELDO
    FROM EMPLOYEES
    WHERE EMPLOYEE_ID = &ID;
    
    IF V_SUELDO >= 10000 THEN
       V_MENSAJE := 'SUELDO ALTO';
    ELSIF V_SUELDO >= 7000 AND V_SUELDO <= 9999 THEN
       V_MENSAJE := 'BUEN SUELDO';
    ELSIF V_SUELDO >= 4000 AND V_SUELDO <= 6999 THEN
       V_MENSAJE := 'SUELDO BAJO';
    ELSE
       V_MENSAJE := 'F';
    END IF;
    DBMS_OUTPUT.PUT_LINE(V_NOMBRE||' '||V_APELLIDO||' -> : '||V_MENSAJE);
END;

/*
    CREE UN BLOUE ANONIMO QUE PERMITA EL INGRESO DEL ID DE UN EMPLEADO,
    DEBE CASIFICAR SI LA COMISION DEL EMPLEAO ES ALTA O BAJA SI ES > 0.3
    ES UNA COMISION ALTA
    SI NO TIENE COMISION CLASIFICA SU SUELDO EN ALTO O BAJO DONDE UN SUEDO
    ALTO DEBE SER MAYOR A 8.000
*/
DECLARE
    V_SUELDO      EMPLOYEES.SALARY%TYPE;
    V_COMMISSION   EMPLOYEES.COMMISSION_PCT%TYPE;
    V_ID_EMPLEADO EMPLOYEES.EMPLOYEE_ID%TYPE := &ID_EMPLEADO;
BEGIN
    SELECT SALARY, COMMISSION_PCT
    INTO V_SUELDO, V_COMMISSION
    FROM EMPLOYEES
    WHERE EMPLOYEE_ID = V_ID_EMPLEADO;
    
    IF V_COMMISSION IS NOT NULL THEN
        IF V_COMMISSION > 0.3 THEN
            DBMS_OUTPUT.PUT_LINE('LA COMISION DEL EMPLEADO ES ALTA');
        ELSE
            DBMS_OUTPUT.PUT_LINE('LA COMISIN DEL MPLEADO ES BAJA');
        END IF;
    ELSE
        IF V_SUELDO > 8000 THEN
            DBMS_OUTPUT.PUT_LINE('EL SUELDO ES ALTO');
        ELSE
            DBMS_OUTPUT.PUT_LINE('EL SUELDO ES BAJO');
        END IF;
    END IF;
END;
