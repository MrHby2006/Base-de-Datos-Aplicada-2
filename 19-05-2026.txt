--EJERCICIO DE CALENTAMIENTO + NUEVA MATERIA (VARIABLES RECORD Y USO DE EXCEPTIONS)
DECLARE

V_ID_EMPLEADO  EMPLOYEES.EMPLOYEE_ID%TYPE := &ID_EMPLEADO;

R_DATOS_EMPLEADO EMPLOYEES%ROWTYPE; --CREA VARIABLIE TIPO RECORD IGUAL A UNA TABLA EN ESPECIFICO

TYPE DATOS_EMPLEADO IS RECORD( --CRECION DE UNA VARIABLE TIPO RECORD
    NOMBRE      VARCHAR(50),
    SUELDO      INTEGER,
    COMMISSION  EMPLOYEES.COMMISSION_PCT%TYPE,
    NOMBRE_DEPA VARCHAR(50)
);

R_DATOS DATOS_EMPLEADO; --CREAR UN OBJETO DEL TIPO DE LA VARIABLE

BEGIN

SELECT E.FIRST_NAME, E.SALARY, E.COMMISSION_PCT, D.DEPARTMENT_NAME
INTO R_DATOS --SE GUARDA E LA VARIABLE, SE USA PARA NO TENER QUE CREAR TANTAS VARIABLES
FROM EMPLOYEES E
JOIN DEPARTMENTS D ON D.DEPARTMENT_ID = E.DEPARTMENT_ID
WHERE EMPLOYEE_ID = V_ID_EMPLEADO;

DBMS_OUTPUT.PUT_LINE('NOBRE EMPLEADO: '||R_DATOS.NOMBRE); --COMO ACCEDER A LOS DATOS DE UNA VARIABLE RECORD
DBMS_OUTPUT.PUT_LINE('SUELDO EMPLEADO: '||R_DATOS.SUELDO);
DBMS_OUTPUT.PUT_LINE('COMISION EMPLEADO: '||R_DATOS.COMMISSION);
DBMS_OUTPUT.PUT_LINE('DEPARTAMENTO DEL EMPLEADO: '||R_DATOS.NOMBRE_DEPA);

EXCEPTION

WHEN NO_DATA_FOUND THEN --EXCEPCION CUANDO NO HAY INFORMACION QUE MOSTRAR
    DBMS_OUTPUT.PUT_LINE('EMPLEADO NO EXISTE');

WHEN TOO_MANY_ROWS THEN --EXCEPCION CUANDO MUSTRA MAS DE UN RESULTADO
    DBMS_OUTPUT.PUT_LINE('LA CONSULTA DEVOLVIO MAS DE UN RESULTADO');
    
WHEN OTHERS THEN --CUALQUIER OTRO ERROR NO PREVISTO
    DBMS_OUTPUT.PUT_LINE('OCURRIO UN ERROR CODIGO: '||SQLCODE);
    DBMS_OUTPUT.PUT_LINE('MENSAJE ERROR: '||SQLERRM);

END;
