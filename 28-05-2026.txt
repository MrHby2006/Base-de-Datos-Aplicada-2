/*
    CREAR UN BLOQUE ANONIMO QUE PERMITA:
    - CALCULAR UN BONO DE LOS EMPLEADOS DE LA EMPRESA
    - SE DEBE RECORRER LA LISTA COMPLETA DE LA TABLA EMPLOYEES, CALIFICANDO:
    - SI EL EMPLEADO TIENE UN SUELDO SUPERIOR A 10.000 -> NO RECIBE BONO
    - SI EL EMPLEADO TIENE UN SUELDO ENTRE 9.999 Y 7.000 -> TENDRA UN BONO DEL 12% DE SU SUELDO
    - SI EL EMPLEADO TIENE UN SUELDO ENTRE 6.999 Y 4.000 -> TENDRA UN BONO DEL 21% DE SU SUELDO
    - SI EL EMPLEADO TIENE UN SUELDO DE 3.000 O INFERIOR -> EPLICA UN BONO DEL 26% DE SU SUELDO
    - SI EL EMPLEADO ES PARTE DEL DEPARTAMENTO 50 O 60 -> AL BONO SE LE DEBE SUMAR 150
    LA SALIDA DEBE MOSTRAR:
    - NOMBRE EMPLEADO: JUANITO PEREZ
    - CORREO: JUANITO@EMPRESA.CL
    - DEPARTAMENTO: 50
    - SUELDO: 11.000
    - BONO: 150
    - SUELDO C/BONO: 11.150
*/
SELECT * FROM EMPLOYEES;

DECLARE
    TYPE RECORD_EMPLEADOS IS RECORD(
        NOMBRE       VARCHAR(20),
        CORREO       VARCHAR(20),
        DEPA         NUMBER, 
        SUELDO       NUMBER
    );
    R_EMPLEADOS RECORD_EMPLEADOS;
    V_BONO        NUMBER;
    V_SUELDO_BONO NUMBER;
    
    V_MIN NUMBER;
    V_MAX NUMBER;
BEGIN
    SELECT MIN(EMPLOYEE_ID), MAX(EMPLOYEE_ID)
    INTO V_MIN, V_MAX
    FROM EMPLOYEES;

    FOR i IN V_MIN..V_MAX
    LOOP
        SELECT FIRST_NAME, EMAIL, DEPARTMENT_ID, SALARY
        INTO R_EMPLEADOS
        FROM EMPLOYEES
        WHERE EMPLOYEE_ID = i;
    
        IF R_EMPLEADOS.SUELDO >= 10000 THEN
            V_BONO := 0;
        ELSIF R_EMPLEADOS.SUELDO >= 7000 AND R_EMPLEADOS.SUELDO <= 9999 THEN
            V_BONO := (R_EMPLEADOS.SUELDO * 12) / 100;
        ELSIF R_EMPLEADOS.SUELDO >= 4000 AND R_EMPLEADOS.SUELDO <= 6999 THEN
            V_BONO := (R_EMPLEADOS.SUELDO * 21) / 100;
        ELSIF R_EMPLEADOS.SUELDO <= 3000 THEN
            V_BONO := (R_EMPLEADOS.SUELDO * 26) / 100;
        ELSE
            V_BONO := 0;
        END IF;
    
        IF R_EMPLEADOS.DEPA IN (60, 50) THEN
            V_BONO := V_BONO + 150;
        END IF;
        
        V_SUELDO_BONO := R_EMPLEADOS.SUELDO + V_BONO;
        
        DBMS_OUTPUT.PUT_LINE('----------------------------');
        DBMS_OUTPUT.PUT_LINE('NOMBRE EMPLEADO: '||R_EMPLEADOS.NOMBRE);
        DBMS_OUTPUT.PUT_LINE('CORREO: '||R_EMPLEADOS.CORREO||'@EMPRESA.CL');
        DBMS_OUTPUT.PUT_LINE('DEPARTAMENTO: '||R_EMPLEADOS.DEPA);
        DBMS_OUTPUT.PUT_LINE('SUELDO: $'||R_EMPLEADOS.SUELDO);
        DBMS_OUTPUT.PUT_LINE('BONO: $'||V_BONO);
        DBMS_OUTPUT.PUT_LINE('SUELDO CON BONO: $'||V_SUELDO_BONO);
    END LOOP;
END;
